```python
    query = state["query"]
    messages = state.get("messages", [])

    logger.info(f"[IntentNode] Processing query: {query}")
    logger.info(f"[IntentNode] Current messages count: {len(messages)}")

    # Check if this is a clarification follow-up
    is_clarification_followup = False
    if len(messages) >= 2:
        # Check if the last assistant message was a clarification question
        last_assistant_msg = None
        for msg in reversed(messages):
            if msg.get("role") == "assistant":
                last_assistant_msg = msg
                break

        if last_assistant_msg and last_assistant_msg.get("type") == "clarification":
            is_clarification_followup = True
            logger.info(f"[IntentNode] Detected clarification follow-up")

            # Merge query with history context
            merged_query = _merge_with_history(messages, query)
            logger.info(f"[IntentNode] Merged query: {merged_query}")
            query = merged_query

    # ... 调用LLM时传递历史上下文 ...
```

新增辅助函数：
```python
def _merge_with_history(messages: list, current_query: str) -> str:
    """Merge current query with conversation history context."""
    previous_user_msg = None

    for msg in reversed(messages):
        if msg.get("role") == "user" and msg.get("content") != current_query:
            previous_user_msg = msg.get("content")
            break

    if previous_user_msg:
        merged = f"{previous_user_msg}，补充信息：{current_query}"
        logger.info(f"[IntentNode] Merged context: {merged}")
        return merged

    return current_query
```

关键改进：
- ✅ 检测澄清回复（assistant.type=clarification）
- ✅ 从历史中提取前一条user消息
- ✅ 合并意图："分析苹果股价" + "补充信息：近期的"
- ✅ 添加日志验证消歧逻辑

```python
def get_intent_prompt(user_query: str, api_list: str = "", history: list = None) -> str:
    """Get intent recognition prompt with optional conversation history."""

    # Build conversation history section
    history_section = ""
    if history and len(history) > 0:
        recent_history = history[-6:]  # 最近6轮对话
        history_lines = []
        for msg in recent_history:
            role = "用户" if msg.get("role") == "user" else "助手"
            content = msg.get("content", "")
            history_lines.append(f"{role}: {content}")
        history_section = f"""

### 对话历史
以下是最近的对话记录，请结合历史上下文理解用户完整意图：
{chr(10).join(history_lines)}
"""

    return INTENT_PROMPT.format(user_query=user_query, api_list=api_list) + history_section
```

关键改进：
- ✅ 添加history参数（可选）
- ✅ 格式化最近6轮对话（3个来回）
- ✅ 在Prompt末尾添加"### 对话历史"字段

**修复1: chat.py - 正确保存user消息和最新assistant响应**

```python
# Step 1: 添加user当前请求（CRITICAL）
user_msg = {
    "role": "user",
    "content": request.message,
    "timestamp": datetime.now().isoformat()
}
session["messages"].append(user_msg)

# Step 2: 只保存LangGraph返回的最后一条assistant消息
if messages:
    last_msg = messages[-1]  # ✅ 只取最后一条（新响应）
    if isinstance(last_msg, dict) and last_msg.get("role") == "assistant":
        session_msg = {
            "role": "assistant",
            "content": last_msg.get("content"),
            "timestamp": datetime.now().isoformat()
        }
        if "type" in last_msg:
            session_msg["type"] = last_msg["type"]  # ✅ 保留type字段
        session["messages"].append(session_msg)
```
